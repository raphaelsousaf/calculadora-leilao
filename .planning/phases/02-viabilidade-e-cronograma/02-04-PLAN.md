       const parseThreshold = (s) => {

         const n = Number(trimmed)
         return Number.isFinite(n) ? n : NaN
       }
       const t1 = parseThreshold(t1Str)
       const t2 = parseThreshold(t2Str)
       const t3 = parseThreshold(t3Str)

       // Cada um isoladamente: NaN ou fora de (0, 100] -> erro
       const provided = [t1, t2, t3]
       if (provided.some(v => Number.isNaN(v))) {
         setThresholdError('Thresholds devem ser numéricos.')
         return
       }
       const allFilled = t1 != null && t2 != null && t3 != null
       const noneFilled = t1 == null && t2 == null && t3 == null
       if (!allFilled && !noneFilled) {
         setThresholdError('Informe os três thresholds ou deixe todos vazios para usar padrões (70/85/95).')
         return
       }
       if (allFilled) {
         if (!(t1 > 0 && t2 > t1 && t3 > t2 && t3 <= 100)) {
           setThresholdError('Os thresholds devem satisfazer 0 < T1 < T2 < T3 ≤ 100.')
           return
         }
       }
       setThresholdError('')
