    ```js
    const DAY_MS = 86400000

    export function addDaysISO(iso, days) {
      if (!iso || typeof iso !== 'string') return ''
      const [y, m, d] = iso.split('-').map(Number)
      if (!y || !m || !d) return ''
      // Constrói em UTC para evitar drift por timezone local
      const base = Date.UTC(y, m - 1, d)
      const next = new Date(base + days * DAY_MS)
      const yy = next.getUTCFullYear()
      const mm = String(next.getUTCMonth() + 1).padStart(2, '0')
      const dd = String(next.getUTCDate()).padStart(2, '0')
      return `${yy}-${mm}-${dd}`
    }

    /**
     * Gera cronograma de N parcelas a partir da data do leilão.
     * @param {number} installments  N parcelas (>=0)
     * @param {string} dataLeilaoISO YYYY-MM-DD (ou '' / null → retorna [])
     * @param {number} intervaloDias intervalo entre parcelas (default 30, clamp 1..365)
     * @param {number} valorParcela  valor BRL de cada parcela (default 0)
     * @returns {{n:number, dueDate:string, amount:number}[]}
     */
    export function buildSchedule(installments, dataLeilaoISO, intervaloDias = 30, valorParcela = 0) {
      const N = Math.max(0, Math.floor(Number(installments) || 0))
      if (N === 0) return []
      if (!dataLeilaoISO) return []

      let interval = Number(intervaloDias)
      if (!Number.isFinite(interval) || interval < 1) interval = 30
      if (interval > 365) interval = 365

      const amount = Number.isFinite(Number(valorParcela)) ? Number(valorParcela) : 0

      const out = []
      for (let i = 1; i <= N; i++) {
        out.push({
          n: i,
          dueDate: addDaysISO(dataLeilaoISO, i * interval),
          amount,
        })
      }
      return out
    }
    ```
